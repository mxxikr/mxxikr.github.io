X = np.array([[51,55], [14,19], [0,4]])
print(X)
# [[51 55]
#  [14 19]
#  [ 0  4]]
X[0] # array([51, 55])
X[0][1] # array([55])
X[0,1] # 55

Y = X.flatten()
print(Y)
# [51 55 14 19 0 4]

for row in X:
    print(row)
# [51 55]
# [14 19]
# [0 4]

Y > 15
# array([True, True, False, True, False, False])
Y[Y > 15]
# array([51, 55, 19])
