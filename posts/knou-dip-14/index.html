X = np.array([[51,55], [14,19], [0,4]])
print(X)
# [[51 55]
#  [14 19]
#  [ 0  4]]
X[0] # array([51, 55])
X[0][1] # array([55])
X[0,1] # 55

Y = X.flatten()
print(Y)
# [51 55 14 19 0 4]

for row in X:
    print(row)
# [51 55]
# [14 19]
# [0 4]

Y > 15
# array([True, True, False, True, False, False])
Y[Y > 15]
# array([51, 55, 19])

import numpy as np

x = np.arange(0, 6, 0.1)
y1 = np.sin(x)
y2 = np.cos(x)
plt.plot(x, y1, label='sin')

plt.xlabel('x-axis')
plt.ylabel('y-axis')
plt.title('sin & cos')
plt.legend()
plt.show()
